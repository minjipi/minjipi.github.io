4) 서비스 설정

[1] controller

mv /etc/cinder/cinder.conf /etc/cinder/cinder.conf.org vi /etc/cinder/cinder.conf
[DEFAULT] my_ip = 컨트롤러 IP log_dir = /var/log/cinder state_path = /var/lib/cinder auth_strategy = keystone transport_url = rabbit://openstack:비번@컨트롤러 IP enable_v3_api = True [database] connection = mysql+pymysql://cinder:비번@컨트롤러IP/cinder [keystone_authtoken] www_authenticate_uri = http://컨트롤러IP:5000 auth_url = http://컨트롤러IP:5000 memcached_servers = 컨트롤러IP:11211 auth_type = password project_domain_name = default user_domain_name = default project_name = service username = cinder password = 비번 [oslo_concurrency] lock_path = /var/lib/cinder/tmp

​

chmod 640 /etc/cinder/cinder.conf chgrp cinder /etc/cinder/cinder.conf

su -s /bin/bash cinder -c "cinder-manage db sync" systemctl enable openstack-cinder-api openstack-cinder-scheduler systemctl restart openstack-cinder-api openstack-cinder-scheduler echo "export OS_VOLUME_API_VERSION=3" >> ~/keystone_admin source ~/keystone_admin

cinder (블록 스토리지 서비스) 설치

'컴퓨트노드' 설치 및 설정

🖥

= = = = = = = = = = = compute 노드 = = = = = = = = = =

1) 패키지 설치

dnf --enablerepo=centos-openstack-victoria,powertools -y install openstack-cinder targetcli

​

2) 서비스 설정

mv /etc/cinder/cinder.conf /etc/cinder/cinder.conf.org vi /etc/cinder/cinder.conf
[DEFAULT] my_ip = '컴퓨트' 아이피 !!!!!!!!! log_dir = /var/log/cinder state_path = /var/lib/cinder auth_strategy = keystone transport_url = rabbit://openstack:비번@컨 트 롤 러 아 이 피 glance_api_servers = http://컨 트 롤 러 아 이 피:9292 enable_v3_api = True enabled_backends = lvm [database] connection = mysql+pymysql://cinder:비번@컨 트 롤 러 아 이 피/cinder [keystone_authtoken] www_authenticate_uri = http://컨 트 롤 러 아 이 피:5000 auth_url = http://컨 트 롤 러 아 이 피:5000 memcached_servers = 컨 트 롤 러 아 이 피:11211 auth_type = password project_domain_name = default user_domain_name = default project_name = service username = cinder password = 비번 [oslo_concurrency] lock_path = /var/lib/cinder/tmp [lvm] target_helper = lioadm target_protocol = iscsi target_ip_address = '컴퓨트' 아이피 !!!!!!!!! volume_group = cinder-volumes volume_driver = cinder.volume.drivers.lvm.LVMVolumeDriver

​

vi /etc/nova/nova.conf

마지막 줄에 추가

[cinder] os_region_name = RegionOne

3) LVM 설정

dnf -y install lvm2 device-mapper-persistent-data
mkdir /var/lib/cinder fallocate -l 300G /var/lib/cinder/cinder-volumes losetup /dev/loop0 /var/lib/cinder/cinder-volumes vgcreate cinder-volumes /dev/loop0
systemctl enable openstack-cinder-volume.service target.service systemctl restart openstack-cinder-volume.service target.service
