from django.shortcuts import render, redirect import paho.mqtt.client as mqtt def pub(request): return render(request, 'pub.html') def sub(request): return render(request, 'sub.html') def publish(request): if request.method == "POST": topic = request.POST['topic'] # 토픽 전달받음 message = request.POST['message'] # 메시지 전달받음 mqttc = mqtt.Client("client", transport='websockets') mqttc.connect("192.168.1.198", 9001) mqttc.publish(topic, message, 1) # QoS 1로 publishing return redirect('/mqtt/pub') # 다시 pub.html 페이지 보여줌

4. 라즈베리파이 코드

# -*- coding: utf-8 -*- import paho.mqtt.client as mqtt import RPi.GPIO as GPIO import time button_pin = 15 led_pin=4 GPIO.setwarnings(False) GPIO.setmode(GPIO.BCM) GPIO.setup(button_pin,GPIO.IN, pull_up_down=GPIO.PUD_DOWN) GPIO.setup(led_pin, GPIO.OUT) light_on=False def button_callback(channel): topic = "led/1" message = "on" global light_on if light_on == False: GPIO.output(led_pin,1) mqttc = mqtt.Client("client", transport='websockets') mqttc.connect("192.168.1.198", 9001) mqttc.publish(topic, message, 1) else: GPIO.output(led_pin,0) print("OFF") mqttc = mqtt.Client("client", transport='websockets') mqttc.connect("192.168.1.198", 9001) mqttc.publish(topic, "Off", 1) light_on=not light_on GPIO.add_event_detect(button_pin,GPIO.RISING, callback=button_callback) while 1: time.sleep(0.1)
