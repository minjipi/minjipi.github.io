# 라즈베리파이로 조이스틱 컨트롤하기

# -*- coding: utf-8 -*- import time import spidev delay = 0.5 sw_channel=0 vrx_channel=1 vry_channel=2 spi=spidev.SpiDev() spi.open(0,0) spi.max_speed_hz=100000 def readadc(adcnum): if adcnum > 7 or adcnum < 0: return -1 r = spi.xfer([1,8+adcnum<<4,0]) data=((r[1]&3)<<8)+r[2] return data while True: vrx_pos = readadc(vrx_channel) vry_pos = readadc(vry_channel) sw_val = readadc(sw_channel) print("x:{} Y:{} SW:{}".format(vrx_pos, vry_pos, sw_val)) time.sleep(delay)

라즈베리파이 코드

# -*- coding: utf-8 -*- import time import spidev import paho.mqtt.client as mqtt mqttc = mqtt.Client("client", transport='websockets') mqttc.connect("[설정한 IP]", 9001) mqttc.publish("minji", "minji", 1) delay = 0.5 sw_channel = 0 vrx_channel = 1 vry_channel = 2 spi = spidev.SpiDev() spi.open(0, 0) spi.max_speed_hz = 100000 def readadc(adcnum): if adcnum > 7 or adcnum < 0: return -1 r = spi.xfer([1, 8 + adcnum << 4, 0]) data = ((r[1] & 3) << 8) + r[2] return data qos = 1 up = False down = False left = False right = False i = 0 # 무한루프 while True: vrx_pos = readadc(vrx_channel) vry_pos = readadc(vry_channel) sw_val = readadc(sw_channel) if vrx_pos > 920: if right == False: print("right") mqttc.publish("minji", "right", qos) up = False down = False left = False right = True i += 1 if vrx_pos < 100: if left == False: print("left") mqttc.publish("minji", "left", qos) up = False down = False left = True right = False i += 1 if vry_pos > 920: if up == False: print("up") mqttc.publish("minji", "up", qos) up = True down = False left = False right = False i += 1 if vry_pos < 100: if down == False: print("down") mqttc.publish("minji", "down", qos) up = False down = True left = False right = False i += 1 time.sleep(0.1) print("x:{} Y:{} SW:{}".format(vrx_pos, vry_pos, sw_val))
