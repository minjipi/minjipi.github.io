[클래스 인터페이스 구현]

자바 클래스는 부모 클래스 '하나'만 상속이 가능하다.

​

extends를 이용한 클래스 상속은 하나만 되고,

두 개 이상은 인터페이스만 가능.

인터페이스를 implements 하는 것은 상속이 아니라 구현 한다고 한다.

그리고 이렇게 구현 할 때는 아래와 같이 ,콤,마, 로 구분해서 작성.

​

*implements= 인터페이스에 정의된 맴버들을 클래스에서 대신 구현 하겠다!

private class Controller implements ActionListener, ChangeListener { @Override public void actionPerformed(ActionEvent e) { // TODO Auto-generated method stub } @Override public void stateChanged(ChangeEvent e) { // TODO Auto-generated method stub } }

​

또 상속과 구현을 동시에 할 수도 있다.

extend키워드를 통해 클래스 하나와,

implements로 인터페이스를 하나 이상 추가하는 것.

​

또 여기서 콤,마를 사용해 다중 상속도 가능하다.

class Controller extends JPanel implements ActionListener, ChangeListener { @Override public void actionPerformed(ActionEvent e) { // TODO Auto-generated method stub } @Override public void stateChanged(ChangeEvent e) { // TODO Auto-generated method stub }

class ImageService { standingImageName = { "f.png" }; walkingLeftImageName = { "l_1.png", ~ "l_10.png" }; walkingRightName = { "r_1.png", ~ "r_10.png" }; } // move 추가 abstract class MovingStrategy { abstract public void move(JLabel lbl); } //서있어! class StandingStrategy extends MovingStrategy { @Override public void move(JLabel lbl) { lbl.setIcon(new ImageIcon("./src/img/" + ImageService.standingImageName[0])); } } //왼쪽으루 가! class WalkingLeftStrategy extends MovingStrategy { private int index = 0; @Override public void move { if (ImageService.walkingLeftImageName.length <= index) { index = 0; } lbl.setIcon(new ImageIcon("./src/img/" + ImageService.walkingLeftImageName[index++])); } } //오른쪽으루 가 class WalkingRightStrategy extends MovingStrategy { private int index = 0; @Override public void move(JLabel lbl) { if (ImageService.walkingRightName.length <= index) { index = 0; } lbl.setIcon(new ImageIcon("./src/img/" + ImageService.walkingRightName[index++])); } } class Zelda extends JFrame { //컨트롤 버튼 만들어 주기 private Controller controller; private MovingStrategy strategy = new StandingStrategy(); public Zelda() { setLayout(new BorderLayout()); field = new Field(); //만든 컨트롤러를 추가해준다 controller = new Controller(); add(controller, BorderLayout.SOUTH); // add(field, BorderLayout.CENTER); Thread th = new Thread(field); th.start(); } private class Field extends JPanel implements Runnable { //쓰레드 구현. public Field() { setLayout(new BorderLayout()); lbl = new JLabel(); // add(lbl); } public void run() } private class Controller extends JPanel implements ActionListener { // 액션 리스너. 액션을 받아 오는 스레드. private JButton btnStanding, btnWalkingLeft, btnWalkingRight; public Controller() { setLayout(new FlowLayout()); btnStanding = new JButton("Standing"); btnWalkingLeft = new JButton("btnWalkingLeft"); btnWalkingRight = new JButton("btnWalkingRight"); btnStanding.addActionListener(this); //액션 리스너 했을 때 대신 처리해줄 애 등록. 그게 버튼. 다형성이래요!! btnWalkingLeft.addActionListener(this); btnWalkingRight.addActionListener(this); add(btnStanding); add(btnWalkingLeft); add(btnWalkingRight); } @Override public void actionPerformed(ActionEvent e) { // JButton btn = (JButton) e.getSource(); if (btn == btnStanding) { strategy = new StandingStrategy(); } else if (btn == btnWalkingLeft) { strategy = new WalkingLeftStrategy(); } else if (btn == btnWalkingRight) { strategy = new WalkingRightStrategy(); } } } }
