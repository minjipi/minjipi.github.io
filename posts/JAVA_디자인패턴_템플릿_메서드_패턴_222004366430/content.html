해결책

- 공통되는 부분을 빼서 abstract 클래스 만들기

abstract class Motor { protected Door door; private MotorStatus motorStatus; public Motor(Door door) { this.door = door; motorStatus = motorStatus.STOPPED; } public MotorStatus getMotorStatus() { return motorStatus; } protected void setMotorStatus(MotorStatus motorStatus) { this.motorStatus = motorStatus; } }

​

하위 클래스

class HD extends Motor { public HD(Door door) { super(door); } private void moveHD(Direction direction) { // 현대 모터 구동 시키는 코드 } public void move(Direction direction) { MotorStatus motorStatus = getMotorStatus(); if (motorStatus == motorStatus.MOVING) return; DoorStatus doorStatus = door.getDoorStatus(); if (doorStatus == doorStatus.OPENED) door.close(); moveHD(direction); setMotorStatus(MotorStatus.MOVING); } } // class LG extends Motor { public LG(Door door) { super(door); } private void moveLG(Direction direction) { // LG 모터 구동 시키는 코드 } public void move(Direction direction) { MotorStatus motorStatus = getMotorStatus(); if (motorStatus == motorStatus.MOVING) return; DoorStatus doorStatus = door.getDoorStatus(); if (doorStatus == doorStatus.OPENED) door.close(); moveLG(direction); setMotorStatus(MotorStatus.MOVING); } }

해 결 책

- move 메서드에서 공통적인 부분을 상위 클래스 Motor로 이동한다!

​

abstract class Motor { protected Door door; private MotorStatus motorStatus; public Motor(Door door) { this.door = door; motorStatus = motorStatus.STOPPED; } public MotorStatus getMotorStatus() { return motorStatus; } protected void setMotorStatus(MotorStatus motorStatus) { this.motorStatus = motorStatus; } public void move(Direction direction) { MotorStatus motorStatus = getMotorStatus(); if (motorStatus == motorStatus.MOVING) return; DoorStatus doorStatus = door.getDoorStatus(); if (doorStatus == doorStatus.OPENED) door.close(); moveMotor(direction); setMotorStatus(MotorStatus.MOVING); } protected abstract void moveMotor(Direction direction); }

그리고 각각의 부분 코드.

class HD extends Motor { public HD(Door door) { super(door); } protected void moveMotor(Direction direction) { // 현대 모터 구동 시키는 코드 } } class LG extends Motor { public LG(Door door) { super(door); } protected void moveMotor(Direction direction) { // LG 모터 구동 시키는 코드 } }
