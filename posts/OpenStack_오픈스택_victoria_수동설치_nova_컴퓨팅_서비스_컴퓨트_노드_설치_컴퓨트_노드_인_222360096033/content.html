1) Compute 노드 서비스 설치

[2] compute

dnf -y install centos-release-openstack-victoria sed -i -e "s/enabled=1/enabled=0/g" /etc/yum.repos.d/CentOS-OpenStack-victoria.repo dnf --enablerepo=centos-openstack-victoria -y upgrade dnf config-manager --set-enabled powertools dnf --enablerepo=centos-openstack-victoria,powertools -y install openstack-nova-compute

​

2) compute 노드 서비스 설정

[2] compute

mv /etc/nova/nova.conf /etc/nova/nova.conf.org
vi /etc/nova/nova.conf
[DEFAULT] enabled_apis = osapi_compute,metadata transport_url = rabbit://openstack:qwer1234@컨트롤러 IP my_ip = 컴.퓨.트.노.드.IP state_path = /var/lib/nova [api] auth_strategy = keystone [keystone_authtoken] www_authenticate_uri = http://컨트롤러 IP:5000/ auth_url = http://컨트롤러 IP:5000/ memcached_servers = 컨트롤러 IP:11211 auth_type = password project_domain_name = Default user_domain_name = Default project_name = service username = nova password = 비번 [vnc] enabled = true server_listen = 0.0.0.0 server_proxyclient_address = 컴.퓨.트.노.드.IP novncproxy_base_url = http://컨트롤러 IP:6080/vnc_auto.html [glance] api_servers = http://controller:9292 [oslo_concurrency] lock_path = /var/lib/nova/tmp [placement] region_name = RegionOne project_domain_name = Default project_name = service auth_type = password user_domain_name = Default auth_url = http://컨트롤러 IP:5000/v3 username = placement password = 비번 [libvirt] virt_type = qemu
chmod 640 /etc/nova/nova.conf chgrp nova /etc/nova/nova.conf
systemctl enable libvirtd.service openstack-nova-compute.service systemctl restart libvirtd.service openstack-nova-compute.service
