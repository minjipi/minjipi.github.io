numbers = [1, 2, 3, 4, 5, 6, 2, 1, 3, 2, 6, 6, 4, 2] counter = {} for key in numbers: if key not in counter: counter[key] = 1 else: counter[key] += 1 print(counter)

​

numbers 리스트의 숫자들이 key값.

리스트 안에서 새로운 숫자가 있으면 그 숫자를 counter의 키값으로 만들고 값을 1로 설정.

새로운 숫자가 아니면 그 숫자 키값에 +1

#메모화

memo = { 1: 1, 2: 1 } def fibo(n): if n in memo: return memo[n] memo[n] = fibo(n - 1) + fibo(n - 2) return memo[n] print(fibo(10)) print(memo)
def flatten(data): flalist = [] for i in data: if type(i) == list: a = flatten(i) # a는 임시 저장 변수. for j in a: flalist.append(j) # 임시 변수에 있던 리스트를 풀어줌. else: flalist.append(i) return flalist ex = [[1, 2, 3], [4, [5, 6]], 7, [8, 9]] print(flatten(ex))
