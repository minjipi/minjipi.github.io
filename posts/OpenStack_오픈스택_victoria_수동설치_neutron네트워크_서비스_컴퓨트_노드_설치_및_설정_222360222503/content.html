[OpenStack]

오픈스택 수동설치

-

neutron(네트워크 서비스)

컴퓨트 노드 설치 및 설정

​

​

1) 서비스 설치

[2] compute

dnf --enablerepo=centos-openstack-victoria,powertools -y install openstack-neutron openstack-neutron-ml2 openstack-neutron-openvswitch

​

2) 서비스 설정

[2] compute

mv /etc/neutron/neutron.conf /etc/neutron/neutron.conf.org
vi /etc/neutron/neutron.conf
[DEFAULT] core_plugin = ml2 service_plugins = router auth_strategy = keystone state_path = /var/lib/neutron allow_overlapping_ips = True transport_url = rabbit://openstack:비번@컨트롤러 IP [keystone_authtoken] www_authenticate_uri = http://컨트롤러 IP:5000 auth_url = http://컨트롤러 IP:5000 memcached_servers = 컨트롤러 IP:11211 auth_type = password project_domain_name = default user_domain_name = default project_name = service username = neutron password = 비번 [oslo_concurrency] lock_path = /var/lib/neutron/lock

저장 후 나와서,

​

chmod 640 /etc/neutron/neutron.conf chgrp neutron /etc/neutron/neutron.conf

​

vi /etc/neutron/plugins/ml2/ml2_conf.ini

위 파일 마지막 줄에 추가

[ml2] type_drivers = flat,vlan,gre,vxlan tenant_network_types = mechanism_drivers = openvswitch extension_drivers = port_security

​

vi /etc/neutron/plugins/ml2/openvswitch_agent.ini

위 파일 마지막 줄에 추가

[securitygroup] firewall_driver = openvswitch enable_security_group = true enable_ipset = true

​

vi /etc/nova/nova.conf

[DEFAULT] 색션에 아래 내용 추가

use_neutron = True linuxnet_interface_driver = nova.network.linux_net.LinuxOVSInterfaceDriver firewall_driver = nova.virt.firewall.NoopFirewallDriver vif_plugging_is_fatal = True vif_plugging_timeout = 300

마지막 줄에 다음 추가

[neutron] auth_url = http://컨트롤러IP:5000 auth_type = password project_domain_name = default user_domain_name = default region_name = RegionOne project_name = service username = neutron password = 비번 service_metadata_proxy = True metadata_proxy_shared_secret = 비번

​

ln -s /etc/neutron/plugins/ml2/ml2_conf.ini /etc/neutron/plugin.ini
systemctl enable openvswitch systemctl restart openvswitch
ovs-vsctl add-br br-int
systemctl restart openstack-nova-compute
systemctl enable neutron-openvswitch-agent systemctl restart neutron-openvswitch-agent

​

3) 확인

[1] controller
