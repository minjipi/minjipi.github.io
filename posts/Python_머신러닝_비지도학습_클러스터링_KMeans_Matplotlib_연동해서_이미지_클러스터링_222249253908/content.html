0. opencv 설치 (파이참, 주피터)

파이참 경우 : pip install opencv-python 주피터의 경우 : conda install opencv

깔아준 후 아래 코드를 실행해보면 :

import cv2 import numpy as np import urllib.request import matplotlib.pyplot as plt request = urllib.request.urlopen( 'https://movie-phinf.pstatic.net/20200901_160/1598948570961a9BWq_JPEG/movie_image.jpg') image_array = np.asarray(bytearray(request.read()), dtype=np.uint8) image = cv2.imdecode(image_array, cv2.COLOR_BGR2LAB) x, y, z = cv2.split(image) fig = plt.figure(figsize=(10, 7)) ax = plt.axes(projection="3d") ax.scatter3D(x, y, z, marker='o', facecolors=cv2.cvtColor(image, cv2.COLOR_BGR2RGB).reshape(-1, 3) / 255., alpha=0.1) plt.show()

from sklearn.cluster import KMeans image = image.reshape(-1,3) k = 5 clt = KMeans(n_clusters=k) clt.fit(image) print(image) ## 중심점 그래프 ## k_means_predicted = clt.predict(image) centroids = clt.clustercenters plt.figure(figsize = (10, 5)) ax = plt.axes(projection = '3d') ax.scatter(centroids[0,0],centroids[0,1],centroids[0,2] ,c=centroids[0]/255., s=100, label='centroid1') ax.scatter(centroids[1,0],centroids[1,1],centroids[1,2] ,c=centroids[1]/255., s=100, label='centroid1') ax.scatter(centroids[2,0],centroids[2,1],centroids[2,2] ,c=centroids[2]/255., s=100, label='centroid1') ax.scatter(centroids[3,0],centroids[3,1],centroids[3,2] ,c=centroids[3]/255., s=100, label='centroid1') ax.scatter(centroids[4,0],centroids[4,1],centroids[4,2] ,c=centroids[4]/255., s=100, label='centroid1') ax.legend() plt.show()
